## 产品概述

理发预约功能，允许全体用户在指定时段预约理发服务，招待员提供服务。

## 核心功能

### 时段规则

- 服务时间：每周一、三、五，下午 14:30 - 18:00
- 时段划分：每半小时一档，显示格式为"14:30~15:00"：
- `14:30~15:00`、`15:00~15:30`、`15:30~16:00`、`16:00~16:30`、`16:30~17:00`、`17:00~17:30`、`17:30~18:00`
- 共 7 档/天
- 节假日排除：自动读取 holiday_configs 集合，排除已配置的节假日

### 预约功能

- **预约权限**：全体已注册用户均可预约
- **预约对象**：
- 可为自己预约（显示：姓名）
- 可为他人预约（显示：理发人姓名（代约人姓名））
- **预约信息**：理发人姓名（必填）
- **时段限制**：每个时段仅限 1 人预约
- **时段禁用规则**：
- 理发日当日下午 14:20 后，当日全部时段禁用，不再允许预约
- **推送通知**：用户预约成功后，招待员收到推送消息

### 查看功能

- **查看权限**：岗位为"招待员""会计主管""会计""内聘"的用户可查看本月理发预约情况
- 其他人不可查看
- 支持两种排序方式：按时间顺序、按人员顺序
- 支持查看历史预约记录

### 取消功能

- **用户取消**：预约人可取消自己创建的预约
- **招待员取消**：招待员可取消他人预约，需选择取消原因：
- 当日招待员因有事未能理发
- 预约人没来理发
- 取消后时段释放，可重新预约

## 技术栈

- 前端框架：微信小程序原生框架
- 数据库：CloudBase NoSQL 文档数据库
- 后端服务：CloudBase 云函数

```javascript
const TIME_SLOTS = [

  { start: '17:00', end: '17:30', display: '17:00~17:30' },
  { start: '17:30', end: '18:00', display: '17:30~18:00' }

// 判断某天是否为理发日
function isHaircutDay(dateStr, holidays) {
  const date = utils.parseLocalDate(dateStr)
  const dayOfWeek = date.getDay() // 0=周日, 1=周一, ...
  
  // 检查是否为周一、三、五
  if (![1, 3, 5].includes(dayOfWeek)) return false
  
  // 检查是否为节假日
  if (holidays.includes(dateStr)) return false
  
  return true
}

// 判断某天是否禁用全部时段（当日14:20后）
function isDayFullyDisabled(dateStr) {
  const now = new Date()
  const targetDate = utils.parseLocalDate(dateStr)
  
  // 检查是否为当天
  if (utils.formatDateObj(now) !== dateStr) return false
  
  // 检查当前时间是否已过14:20
  const currentHour = now.getHours()
  const currentMinute = now.getMinutes()
  const currentTime = currentHour * 60 + currentMinute
  const disableTime = 14 * 60 + 20 // 14:20
  
  return currentTime >= disableTime
}

// 获取可显示的日期列表（最多显示一周的一三五）
function getDisplayableDates(holidays) {
  const dates = []
  const now = new Date()
  const todayStr = utils.formatDateObj(now)
  
  // 判断是否应该显示下周（周五18:00后）
  const shouldShowNextWeek = (() => {
    const dayOfWeek = now.getDay() // 0=周日, 5=周五
    if (dayOfWeek !== 5) return false
    const currentHour = now.getHours()
    return currentHour >= 18
  })()
  
  // 计算起始日期
  let startDate = new Date(now)
  if (shouldShowNextWeek) {
    // 从下周一（+3天）或下周三（+1天）或下周五（-1天）开始
    startDate.setDate(startDate.getDate() + 3) // 跳到下周
    startDate.setHours(0, 0, 0, 0)
  }
  
  // 遍历接下来的14天，找出最多7个理发日
  const tempDate = new Date(startDate)
  while (dates.length < 7) {
    const dateStr = utils.formatDateObj(tempDate)
    const dayOfWeek = tempDate.getDay()
    
    // 只收集周一、三、五
    if ([1, 3, 5].includes(dayOfWeek)) {
      const isHoliday = holidays.includes(dateStr)
      dates.push({
        date: dateStr,
        dayOfWeek: ['日', '一', '二', '三', '四', '五', '六'][dayOfWeek],
        isHoliday,
        isToday: dateStr === todayStr,
        isDisabled: isHoliday || isDayFullyDisabled(dateStr)
      })
    }
    
    tempDate.setDate(tempDate.getDate() + 1)
  }
  
  return dates
}
```

```javascript

function isDaySlotsDisabled(dateStr) {
  const now = new Date()
```

#### 推送通知逻辑
